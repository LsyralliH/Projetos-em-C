        printf("\nDigite o email:\n");

        arroba = ponto = antesPonto = depoisPonto = 0;
        for (i = 0; i < strlen(agenda[ad].email); i++) {
            char c = agenda[ad].email[i];

                if (arroba)
                    break;

                if (i < 3)
                    break;
            } else if (arroba) {
                if (ponto) {
                    depoisPonto++;
                } else if (c == '.') {
                    ponto = 1;
                    if (antesPonto < 3) {
                        break;
                    }
                } else {
                    antesPonto++;
                }
            }
        }
    } while (i != strlen(agenda[ad].email) || depoisPonto < 1);
